cat << 'EOF' > convenzioni.txt
CONVENZIONI MESSAGGI COMMIT:

feat:    Aggiunta di una nuova funzionalita' (Feature)
fix:     Correzione di un bug o di un errore (Fix)
docs:    Modifiche alla documentazione (es. README.md)
style:   Modifiche alla formattazione o CSS (senza logica)
refactor:Riscrittura di codice esistente per migliorarlo
EOF

git add convenzioni.txt
git commit -m "docs: aggiunto file guida sulle convenzioni dei commit  "
